// BarberPro – Google Apps Script // SpreadsheetのIDを YOUR_SHEET_ID に置き換えてください const SHEET_ID = 'YOUR_SHEET_ID'; function doPost(e) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const d = JSON.parse(e.postData.contents); const tab = ss.getSheetByName(d.sheet) || ss.insertSheet(d.sheet); if (d.action === 'init') { // ヘッダー行をセット（なければ） if (tab.getLastRow() === 0) tab.appendRow(d.headers); return ok(); } if (d.action === 'append') { tab.appendRow(d.row); return ok(); } if (d.action === 'update') { const rows = tab.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { tab.getRange(i + 1, 1, 1, d.row.length).setValues([d.row]); return ok(); } } tab.appendRow(d.row); // 見つからなければ追加 return ok(); } if (d.action === 'delete') { const rows = tab.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(d.id)) { tab.deleteRow(i + 1); return ok(); } } return ok(); } if (d.action === 'bulkWrite') { tab.clearContents(); tab.appendRow(d.headers); if (d.rows.length) tab.getRange(2, 1, d.rows.length, d.rows[0].length).setValues(d.rows); return ok(); } } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.openById(SHEET_ID); if (e.parameter.ping) return ok(); // 接続テスト用 const tab = ss.getSheetByName(e.parameter.sheet); if (!tab) return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify(tab.getDataRange().getValues())) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function ok() { return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
